  { egrep -i '(seeds|architectures):' update.cfg | tr -d ':' | while read a b 
    do echo for $a in $b ';' do
    done
    echo echo '$seeds-$architectures'
    echo echo '$seeds-recommends-$architectures'
    echo done ';' done
  } | sh | while read fn
  do echo ---- $fn
     diff $fn <(sed -r "$sedprog" $fn) || sed -r -i "$sedprog" $fn
  done
)
